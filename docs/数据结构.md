# 数据结构

> 创建日期：2026-01-17
> 文件位置：`src/赵霞/schema.ts`
> 状态：已实现

## 概述

赵霞项目使用 Zod 库定义数据结构，确保类型安全和数据验证。所有游戏数据存储在 `stat_data` 变量中。

---

## 世界状态

```typescript
世界: {
  // 时间系统
  当前天数: number;        // 1-5
  当前小时: number;        // 0-23
  时间: string;           // "Day 1, 08:00"
  时间戳: number;         // Date.now()
  状态栏需要刷新: boolean;

  // 循环系统
  循环状态: '进行中' | '结局判定' | '已破解';
  当前循环轮数: number;

  // 双重叙事
  已进入过梦境: boolean;   // false=纯爱模式, true=真相模式

  // 游戏阶段
  游戏阶段: '序章' | '日常' | '梦境' | '结局';

  // 梦境入口（内部字段，下划线前缀）
  _梦境入口消息ID?: number;
  _梦境入口天数?: number;
  梦境选择已锁定: boolean;

  // 摘要生成
  待生成摘要?: {
    sceneNum: number;
    dreamEntryId: number;
    dreamExitId?: number;
  };
  上一轮梦境已退出?: { ... };

  // ROLL检测
  _梦境退出记录?: { ... };
  _梦境入口记录?: { ... };
}
```

## 梦境数据

```typescript
梦境数据: {
  // 当前梦境
  当前记忆年龄?: number;   // 16-40
  梦境心理活动?: string;
  此次梦境目标?: string;
  记忆背景故事?: string;

  // 场景完成
  已完成场景: number[];    // [1, 2, 3, 4, 5]
  正确重构场景: number[];

  // 场景1-4数据（结构相同）
  场景1~4?: {
    已进入: boolean;
    选择部位: string[];
    是否正确?: boolean;
    进入时间?: string;
    对话轮数: number;
    剧情摘要?: string;
  };

  // 场景5（特殊）
  场景5?: {
    // 基础
    已进入: boolean;
    选择部位: string[];
    是否正确?: boolean;
    进入时间?: string;
    对话轮数: number;
    需要安眠药: boolean;

    // 12步系统
    进入次数: number;
    当前步骤: number;      // 0-12
    已完成步骤: boolean;
    上次剧情摘要?: string;
    已植入暗示: string[];

    // 完成度
    完成度: number;        // 0-100
    步骤进度记录: number[];
    可触发特殊结局: boolean;

    // ROLL检测
    上次推进记录?: { ... };
  };

  // 威胁数值
  记忆混乱度: number;      // 0-100

  // 混乱结局
  混乱结局: {
    已标记: boolean;
    标记时间: number | null;
    触发时间: number | null;
    已触发: boolean;
    触发原因: '性行为' | '打断仪式' | null;
    性行为次数: number;
  };

  // 记忆连贯性
  记忆连贯性: number;      // 0-3
  场景5进入时连贯性?: number;

  // 每晚进度
  当晚进度记录: {
    天数: number;
    嘴巴: number;
    胸部: number;
    下体: number;
    后穴: number;
    精神: number;
  };
}
```

## 境界与依存度对应

| 依存度 | 境界 | 纯爱名称 | 真相名称 |
|--------|------|---------|---------|
| 0-19 | 1 | 初识 | 初染 |
| 20-39 | 2 | 好友 | 迷途 |
| 40-59 | 3 | 暧昧 | 溺深 |
| 60-79 | 4 | 热恋 | 归虚 |
| 80-100 | 5 | 真爱 | 焚誓 |

---

## 记忆连贯性等级

| 等级 | 条件 | 说明 |
|------|------|------|
| 0 | 直接进入场景5 | 完全陌生 |
| 1 | 完成场景1 | 有初恋记忆 |
| 2 | 完成场景1+2 | 有初恋+等待记忆 |
| 3 | 完成场景1+2+3 | 完美记忆路线 |
